gcc tsk.c -o tsk.e -g -O0

gdb ./tsk.e

gdb> start
### повторяем эти действия, следим за тем, как меняется битовое представление числа

gdb> x/t &f ### печать битового представления f

set var f = f / 2.0 ### изменение переменной

break 30 ### точка остановки на 30 строке программы
